Mensagens Cliente -> Servidor:
	Formato: COMANDO:DADOS
	
	INICIAR_QUIZ:<quiz><nome> 	Ex: INICIAR_QUIZ:REDES_C1 // iniciar quiz capitulo 1
	REPONDER_PERGUNTA<numero><alternativa> Ex: RESPOSTA:3:A // reponder pergunta do quiz
	PEDIR_PONTUACAO
	PEDIR_RANKING

Mensagens Servidor -> Cliente:
	Formato: COMANDO:DADOS

	BEM_VINDO
	FIM_DE_JOGO
	PERGUNTA:<numero><texto><letraA><letraB><letraC><letraD>
	RESULTADO_CORRETO
	RESULTADO_INCORRETO
	PONTUACAO_FINAL:<pontos>
	RANKING:<pos1_nome>:<pos1_pontos>;<pos2_nome>:<pos2_pontos>;...
